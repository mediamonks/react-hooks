# unwrapRefs

Utility to unwrap RefObjects in an object

## Reference

```ts
export function unwrapRefs<T extends Refs>(refs: T): UnwrapRefs<T>;
```

### Returns

Object with unwrapped RefObjects

## Usage

```tsx
type DemoComponentRefs = Refs<{
  heading: HTMLHeadingElement;
  button: Array<HTMLButtonElement | null>;
}>;

function DemoComponent() {
  const refs = useRef<DemoComponentRefs>();

  const { element, elements } = unwrapRefs(refs);

  console.log(element); // HTMLHeadingElement | null
  console.log(elements); // Array<HTMLButtonElement | null>

  return (
    <div ref={refs.element}>
      {items.map((item, index) => (
        <span key={item} ref={arrayRef(refs.elements, index)}>
          {item}
        </span>
      ))}
    </div>
  );
}
```
