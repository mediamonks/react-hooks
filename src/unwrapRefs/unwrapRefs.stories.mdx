# unwrapRefs

Utility to unwrap RefObjects in an object

## Reference

```ts
export function unwrapRefs<T extends Refs>(refs: T): UnwrapRefs<T>;
```

### Returns

Object with unwrapped RefObjects

## Usage

```tsx
type DemoComponentRefs = Refs<{
  heading: HTMLHeadingElement;
  buttons: Array<HTMLButtonElement | null>;
}>;

function DemoComponent() {
  const refs = useRef<DemoComponentRefs>();

  useEffect(() => {
    const { heading, buttons } = unwrapRefs(refs);

    console.log(heading); // HTMLHeadingElement | null
    console.log(buttons); // Array<HTMLButtonElement | null> | null
  }, [refs]);

  return (
    <>
      <h1 ref={refs.heading}>{heading}</h1>

      {items.map((item, index) => (
        <button key={item} ref={arrayRef(refs.buttons, index)}>
          {item}
        </button>
      ))}
    </>
  );
}
```
